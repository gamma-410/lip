First, download the lip from github / gamma410.

Then place the lip directory under the root of your project.

Add lip.css and lip.js to your project right away.


        <!doctype html>
        <html lang="">
        <head>
            <!-- required meta tags -->
            <meta charset="utf-8">
            <meta name="viewport" content="width=device-width, initial-scale=1">
            

            <!-- lip CSS -->
            <link href="lip/lip.css" rel="stylesheet">
            

            <title>example - Hello, world!</title>
        </head>
        <body>
            <h1>Hello, world!</h1>
            

            <!-- lip JavaScript -->
            <script src="lip/lip.js"></script>
        </body>
        </html>
